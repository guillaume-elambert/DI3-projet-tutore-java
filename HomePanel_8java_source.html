235 
236  currentDate = Calendar.getInstance();
237 
238  // On calcul la différence avec le quart d'heure
239  int mod = currentDate.get(Calendar.MINUTE) % 15;
240 
241  // Si la minute est plus proche du quart d'heure précédent, c'est lui qu'on prend, l'autre sinon
242  currentDate.add(Calendar.MINUTE, (mod < 8 ? -mod : (15 - mod)));
243  currentDate.set(Calendar.SECOND, 0);
244 
245  trueTime.setText(DateFormat.getTimeInstance().format(currentDate.getTime()));
246  }
