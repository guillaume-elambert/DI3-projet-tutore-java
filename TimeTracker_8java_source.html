189  /**

192  *
193  * @return La liste des noms des machines précédemment configurées.

195  public static ArrayList<String> listSettingsFiles()
196  {
197  File root = new File(settingsFolder);
198  ArrayList<String> emuNames = new ArrayList<String>();
199 
200  if (root.isDirectory())
201  {
202  final Pattern p = Pattern.compile("emu-(.*)\\.json"); // careful: could also throw an exception!
203  Matcher matcher;
204 
205  File[] listFiles = root.listFiles();
206 
207  // On parcourt tous les fichiers du dossier ".settings"
208  for (File file : listFiles)
209  {
210  matcher = p.matcher(file.getName());
211 
212  // Si le nom du fichier parcouru correspond au format attendu
213  // => On ajoute son nom à la liste
214  if (matcher.matches())
215  {
216  emuNames.add(new String(Base64.getDecoder().decode(matcher.group(1))));
217  }
218  }
219  }
220 
221  return emuNames;
222  }
